const { MongoClient, ServerApiVersion } = require('mongodb');
require('dotenv').config(); // ‡πÇ‡∏´‡∏•‡∏î‡∏Ñ‡πà‡∏≤‡∏à‡∏≤‡∏Å .env
const uri = `${process.env.MONGO_URL}`
// Create a MongoClient with a MongoClientOptions object to set the Stable API version
const client = new MongoClient(uri, {
  serverApi: {
    version: ServerApiVersion.v1,
    strict: true,
    deprecationErrors: true,
  }
});

exports.connectDB = async () => {
    try {
        await client.connect();
        console.log("‚úÖ MongoDB Connected Successfully");
    } catch (error) {
        console.error("‚ùå MongoDB Connection Error:", error);
        process.exit(1);
    }
};

exports.disconnectDB = async () => {
    try {
        await client.close();
        console.log("‚úÖ MongoDB Disconnected Successfully");
    } catch (error) {
        console.error("‚ùå MongoDB Disconnection Error:", error.message);
    }
};

exports.upsertUserData = async (userId, groupId, data) => {
    if (!userId || !data) {
        throw new Error("‚ùå Missing required parameters: userId and data are required.");
    }

    try {
        const db = client.db();
        const collection = db.collection("user_group");

        const filter = { userId, groupId };
        const updateData = { $set: data };

        const result = await collection.findOneAndUpdate(filter, updateData, {
            returnDocument: "after", // ‚úÖ ‡∏Ñ‡∏∑‡∏ô‡∏Ñ‡πà‡∏≤‡∏Ç‡πâ‡∏≠‡∏°‡∏π‡∏•‡∏ó‡∏µ‡πà‡∏≠‡∏±‡∏õ‡πÄ‡∏î‡∏ï
            upsert: true // ‚úÖ ‡∏™‡∏£‡πâ‡∏≤‡∏á‡πÄ‡∏≠‡∏Å‡∏™‡∏≤‡∏£‡πÉ‡∏´‡∏°‡πà‡∏ñ‡πâ‡∏≤‡∏¢‡∏±‡∏á‡πÑ‡∏°‡πà‡∏°‡∏µ
        });


        await client.close();
        console.log("‚úÖ Upsert Success:", result);
        return result;
    } catch (error) {
        console.error("‚ùå Error in upsertUserData:", error.message);
        throw error;
    }
};


exports.getUserData = async (userId, groupId = null) => {
    if (!userId) {
        throw new Error("‚ùå Missing required parameter: userId is required.");
    }

    try {
        const db = client.db();
        const collection = db.collection("user_group");

        const filter = { userId };
        if (groupId) filter.groupId = groupId;

        const result = await collection.findOne(filter);

        if (!result) {
            console.log(`‚ö†Ô∏è No document found for userId: ${userId}, groupId: ${groupId || "N/A"}`);
            return null;
        }

        console.log("‚úÖ Found User Data:", result);
        await client.close();
        return result;
    } catch (error) {
        console.error("‚ùå Error in getUserData:", error.message);
        throw error;
    }
};


exports.deleteDataByGroupId = async (groupId) => {
    if (!groupId) {
        throw new Error("‚ùå Missing required parameter: groupId is required.");
    }

    try {
        const db = client.db();
        const collection = db.collection("user_group");

        const result = await collection.deleteMany({ groupId });

        console.log(`‚úÖ Deleted ${result.deletedCount} documents in groupId: ${groupId}`);
        await client.close();
        return result;
    } catch (error) {
        console.error("‚ùå Error in deleteDataByGroupId:", error.message);
        throw error;
    } finally {
        await disconnectDB(); // ‚úÖ ‡∏õ‡∏¥‡∏î Connection ‡∏´‡∏•‡∏±‡∏á‡πÉ‡∏ä‡πâ‡∏á‡∏≤‡∏ô
    }
};

exports.deleteDataByGroupAndUserId = async (groupId, userId) => {
    if (!groupId || !userId) {
        throw new Error("‚ùå Missing required parameters: groupId and userId are required.");
    }

    try {
        const db = client.db();
        const collection = db.collection("user_group");

        const result = await collection.deleteOne({ groupId, userId });

        console.log(`‚úÖ Deleted ${result.deletedCount} document(s) in groupId: ${groupId}, userId: ${userId}`);
        return result;
    } catch (error) {
        console.error("‚ùå Error in deleteDataByGroupAndUserId:", error.message);
        throw error;
    } finally {
        await disconnectDB(); // ‚úÖ ‡∏õ‡∏¥‡∏î Connection ‡∏´‡∏•‡∏±‡∏á‡πÉ‡∏ä‡πâ‡∏á‡∏≤‡∏ô
    }
};
